#include <stdio.h>

int main() {
    // Declare variables
    char name[] = "Sayed Hassan Ali";
    int age = 18;
    char class[] = "F.E AI&DS";
    char division = 'A';
    char UIN[] = "241A055";

    // Print output using escape sequences for formatting
    printf("Full Name  : %s\n", name);
    printf("Age        : %d\n", age);
    printf("Class      : %s\n", class);
    printf("Division   : %c\n", division);
    printf("UIN        : %s\n", UIN);

    return 0;
}
